This then allows you to `break` and interrupt the loop when needed – even from inside another loop.

```js
function loop () &lcub;
  dance:
  for (let i = 0; i &lt; 4; i++) &lcub;
    for (let j = 0; j &lt; 4; j++) &lcub;
      if (j === 2) &lcub;
        break dance;
      }
    }
  }
}
